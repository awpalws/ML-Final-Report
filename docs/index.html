human_samples = combined_dataset[combined_dataset['source'] == 0]
ai_samples = combined_dataset[combined_dataset['source'] == 1]

class_sizes = min(len(human_samples), len(ai_samples))

balanced_dataset = pd.concat([human_samples.sample(n=class_sizes, random_state=42), ai_samples.sample(n=class_sizes, random_state=42)]).sample(frac=1, random_state=42)
xtrain, xtest, ytrain, ytest = train_test_split(balanced_dataset['text'],balanced_dataset['source'],test_size=0.3,random_state=42)
